```sql
  user_id       uuid NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,

  grain_week    integer NOT NULL,

ALTER TABLE farm_summaries ENABLE ROW LEVEL SECURITY;

  ON farm_summaries FOR SELECT
  USING (auth.uid() = user_id);

  ON farm_summaries FOR ALL
  USING ((current_setting('role'::text)) = 'service_role');
```

```sql
-- Calculate delivery percentiles across all users for a given crop year and week range.
-- Returns one row per user per grain with their percentile rank.
CREATE OR REPLACE FUNCTION calculate_delivery_percentiles(
  p_crop_year text DEFAULT '2025-2026'
)
RETURNS TABLE (
  user_id uuid,
  grain text,
  total_delivered_kt numeric,
  percentile_rank numeric
)
LANGUAGE sql STABLE
AS $$
  WITH user_totals AS (
    SELECT
      cp.user_id,
      cp.grain,
      COALESCE(
        (SELECT SUM((d->>'amount_kt')::numeric)
         FROM jsonb_array_elements(cp.deliveries) AS d),
        0
      ) AS total_delivered_kt
    FROM crop_plans cp
    WHERE cp.crop_year = p_crop_year
      AND cp.deliveries IS NOT NULL
      AND jsonb_array_length(cp.deliveries) > 0
  )
  SELECT
    ut.user_id,
    ut.grain,
    ut.total_delivered_kt,
    (PERCENT_RANK() OVER (PARTITION BY ut.grain ORDER BY ut.total_delivered_kt)) * 100
      AS percentile_rank
  FROM user_totals ut;
$$;
```

### Task 9: Create generate-farm-summary Edge Function

**Files:**
- Create: `supabase/functions/generate-farm-summary/index.ts`

**Step 1: Write the Edge Function**

```typescript
/**
 * Supabase Edge Function: generate-farm-summary
 *
 * Generates personalized AI farm summaries with delivery percentile comparisons.
 * Chained after generate-intelligence in the weekly pipeline.
 *
 * For each user with active crop_plans:
 * 1. Fetch their deliveries and percentiles
 * 2. Call GPT-4o for a personalized narrative
 * 3. Upsert into farm_summaries
 *
 * Request body (optional):
 *   { "crop_year": "2025-2026", "grain_week": 29 }
 */
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const OPENAI_API_URL = "https://api.openai.com/v1/chat/completions";
const MODEL = "gpt-4o";
const BATCH_SIZE = 50;

Deno.serve(async (req) => {
  const startTime = Date.now();

  try {
    const supabase = createClient(
      Deno.env.get("SUPABASE_URL")!,
      Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
    );

    const openaiKey = Deno.env.get("OPENAI_API_KEY");
    if (!openaiKey) {
      return jsonResponse({ error: "OPENAI_API_KEY not configured" }, 500);
    }

    const body = await req.json().catch(() => ({}));
    const cropYear: string = body.crop_year || getCurrentCropYear();
    const grainWeek: number = body.grain_week || getCurrentGrainWeek();

    console.log(`Generating farm summaries for week ${grainWeek}, crop year ${cropYear}`);

    // Get all users with active crop plans
    const { data: plans, error: plansError } = await supabase
      .from("crop_plans")
      .select("user_id, grain, acres_seeded, volume_left_to_sell_kt, deliveries")
      .eq("crop_year", cropYear);

    if (plansError) {
      return jsonResponse({ error: plansError.message }, 500);
    }

    if (!plans || plans.length === 0) {
      return jsonResponse({ message: "No active crop plans found", count: 0 });
    }

    // Get percentiles for all users
    const { data: percentiles } = await supabase
      .rpc("calculate_delivery_percentiles", { p_crop_year: cropYear });

    const percentileMap: Record<string, Record<string, number>> = {};
    for (const p of percentiles || []) {
      if (!percentileMap[p.user_id]) percentileMap[p.user_id] = {};
      percentileMap[p.user_id][p.grain] = Math.round(p.percentile_rank);
    }

    // Group plans by user
    const userPlans: Record<string, typeof plans> = {};
    for (const plan of plans) {
      if (!userPlans[plan.user_id]) userPlans[plan.user_id] = [];
      userPlans[plan.user_id].push(plan);
    }

    const userIds = Object.keys(userPlans).slice(0, BATCH_SIZE);
    let successCount = 0;
    let errorCount = 0;

    for (const userId of userIds) {
      try {
        const crops = userPlans[userId];
        const userPercentiles = percentileMap[userId] || {};

        const prompt = buildFarmSummaryPrompt(crops, userPercentiles, grainWeek, cropYear);

        const response = await fetch(OPENAI_API_URL, {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            Authorization: `Bearer ${openaiKey}`,
          },
          body: JSON.stringify({
            model: MODEL,
            messages: [
              { role: "system", content: "You are a concise agricultural market analyst writing personalized farm summaries for Canadian prairie farmers. Write 2-4 sentences. Be specific with numbers. Use a warm but professional tone." },
              { role: "user", content: prompt },
            ],
            temperature: 0.7,
            max_tokens: 300,
          }),
        });

        const result = await response.json();
        const summaryText = result.choices?.[0]?.message?.content?.trim();

        if (!summaryText) {
          console.error(`Empty response for user ${userId}`);
          errorCount++;
          continue;
        }

        const { error: upsertError } = await supabase
          .from("farm_summaries")
          .upsert({
            user_id: userId,
            crop_year: cropYear,
            grain_week: grainWeek,
            summary_text: summaryText,
            percentiles: userPercentiles,
            generated_at: new Date().toISOString(),
          }, { onConflict: "user_id,crop_year,grain_week" });

        if (upsertError) {
          console.error(`Upsert error for user ${userId}:`, upsertError.message);
          errorCount++;
        } else {
          successCount++;
        }
      } catch (err) {
        console.error(`Error processing user ${userId}:`, err);
        errorCount++;
      }
    }

    const totalUsers = Object.keys(userPlans).length;
    const remaining = totalUsers - BATCH_SIZE;

    // If more users remain, re-trigger self
    if (remaining > 0) {
      console.log(`${remaining} users remaining, triggering next batch`);
      // Best-effort re-trigger via pg_net (non-blocking)
      await supabase.rpc("trigger_edge_function", {
        function_name: "generate-farm-summary",
        payload: JSON.stringify({ crop_year: cropYear, grain_week: grainWeek }),
      }).catch(() => {
        console.log("Re-trigger not available, remaining users will be processed next run");
      });
    }

    const elapsed = Date.now() - startTime;
    return jsonResponse({
      message: `Generated ${successCount} farm summaries in ${elapsed}ms`,
      success: successCount,
      errors: errorCount,
      total_users: totalUsers,
    });
  } catch (err) {
    return jsonResponse({ error: String(err) }, 500);
  }
});

function buildFarmSummaryPrompt(
  crops: Array<{ grain: string; acres_seeded: number; volume_left_to_sell_kt: number | null; deliveries: any[] | null }>,
  percentiles: Record<string, number>,
  grainWeek: number,
  cropYear: string
): string {
  const cropSummaries = crops.map((c) => {
    const deliveries = c.deliveries || [];
    const totalDelivered = deliveries.reduce((sum: number, d: any) => sum + (d.amount_kt || 0), 0);
    const pct = percentiles[c.grain];
    const pctStr = pct !== undefined ? `(top ${100 - pct}% of users)` : "";
    return `- ${c.grain}: ${c.acres_seeded} acres, delivered ${totalDelivered.toFixed(2)} kt ${pctStr}, ${c.volume_left_to_sell_kt ?? 0} kt remaining to sell`;
  }).join("\n");

  return `Write a personalized farm summary for a prairie farmer in week ${grainWeek} of the ${cropYear} crop year.

Their active crops:
${cropSummaries}

Include: delivery progress highlights, percentile comparisons where available, and any actionable observations. Keep it to 2-4 sentences.`;
}

function getCurrentCropYear(): string {
  const now = new Date();
  const year = now.getFullYear();
  const month = now.getMonth();
  return month >= 7 ? `${year}-${year + 1}` : `${year - 1}-${year}`;
}

function getCurrentGrainWeek(): number {
  const now = new Date();
  const year = now.getFullYear();
  const month = now.getMonth();
  const cropYearStart = month >= 7
    ? new Date(year, 7, 1)
    : new Date(year - 1, 7, 1);
  return Math.ceil((now.getTime() - cropYearStart.getTime()) / (7 * 24 * 60 * 60 * 1000));
}

function jsonResponse(data: unknown, status = 200) {
  return new Response(JSON.stringify(data), {
    status,
    headers: { "Content-Type": "application/json" },
  });
}
```

```typescript
export interface FarmSummary {
  user_id: string;
  crop_year: string;
  grain_week: number;
  summary_text: string;
  percentiles: Record<string, number>;
  generated_at: string;
}

/**
 * Get the latest farm summary for a user.
 */
export async function getFarmSummary(
  userId: string
): Promise<FarmSummary | null> {
  const supabase = await createClient();
  const { data, error } = await supabase
    .from("farm_summaries")
    .select("*")
    .eq("user_id", userId)
    .order("grain_week", { ascending: false })
    .limit(1)
    .single();

  if (error || !data) return null;
  return data as FarmSummary;
}
```
